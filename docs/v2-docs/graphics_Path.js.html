graphics/Path.js

const Graphics = require('./Graphics');

/**
 * A path made of a sequence of points.
 * @extends Graphics
 */
class Path extends Graphics {

  /**
   * Create a path.
   * @param {number} [x=0] The X position (the starting point's x).
   * @param {number} [y=0] The Y position (the starting point's y).
   */
  constructor(x = 0, y = 0) {
    super();

    /**
     * The X position (relative to the game object).
     * @type {number}
     */
    this.x = x;

    /**
     * The Y position (relative to the game object).
     * @type {number}
     */
    this.y = y;

    /**
     * The opacity in percentage.
     * @type {number}
     * @default
     */
    this.opacity = 100;

    /**
     * The width of the strokes.
     * @type {number}
     * @default
     */
    this.lineWidth = 1;

    /**
     * The line cap to use.
     * @type {'butt'|'round'|'square'}
     * @default
     */
    this.lineCap = 'butt';
    
    /**
     * The line join to use.
     * @type {'miter'|'round'|'bevel'}
     * @default
     */
    this.lineJoin = 'miter';

    /**
     * The colour to fill the path with.
     * @type {CanvasTexture}
     * @default
     */
    this.fillStyle = '#ff00ff';

    /**
     * The colour to stroke the path with.
     * @type {CanvasTexture}
     * @default
     */
    this.strokeStyle = '#ffffff';

    /**
     * Whether to fill the path.
     * @type {boolean}
     * @default
     */
    this.fill = true;

    /**
     * Whether to stroke the path.
     * @type {boolean}
     * @default
     */
    this.stroke = false;

    /**
     * The points contained in the path, in the order they were added.
     * @type {Point[]}
     * @default
     */
    this.path = [];
  }

  /**
   * Add a point to the path.
   * @param {number} x The X property of the point.
   * @param {*} y The Y property of the point.
   */
  addPoint(x, y) {
    this.path.push({
      x: x,
      y: y
    });
  }

  /**
   * Render the path.
   * @param {CanvasRenderingContext2D} context The renderer's context.
   * @ignore
   */
  render(context) {
    // Properties
    context.translate(this.x, this.y);
    context.globalAlpha = this.opacity / 100;
    context.lineCap = this.lineCap;
    context.lineJoin = this.lineJoin;
    context.fillStyle = this.fillStyle;
    context.strokeStyle = this.strokeStyle;
    
    // Draw
    context.beginPath();
    context.moveTo(this.path[0].x, this.path[0].y);

    for (let i = 1; i < this.path.length; i++) {
      context.lineTo(this.path[i].x, this.path[i].y);
    }

    context.closePath();

    // Fill/Stroke
    if (this.fill) {
      context.fill();
    }

    if (this.stroke) {
      context.stroke();
    }
  }
}

module.exports = Path;
