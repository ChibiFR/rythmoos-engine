graphics/Circle.js

const Graphics = require('./Graphics');

/**
 * A simple circle.
 * @extends Graphics
 * @example
 * import {Circle} from 'rythmoos-engine';
 * @example <caption>Creating a circle</caption>
 * // Set its center to 50, 50.
 * const myCircle = new Circle(50, 50);
 * @example <caption>Using in a game object</caption>
 * myGameObject.set('my circle', myCircle);
 */
class Circle extends Graphics {

  /**
   * Create a Circle.<br>
   * The point (x, y) represents the center of the circle.
   * @param {number} [x=0] The X position (relative to the game object).
   * @param {number} [y=0] The Y position (relative to the game object).
   * @param {number} [radius=50] The radius of the circle.
   */
  constructor(x = 0, y = 0, radius = 50) {
    super();

    /**
     * The X position (relative to the {@link GameObject|game object}).
     * @type {number}
     * @example
     * // Set myCircle X position to 150.
     * myCircle.x = 150;
     */
    this.x = x;

    /**
     * The Y position (relative to the {@link GameObject|game object}).
     * @type {number}
     * @example
     * // Set myCircle Y position to -30.
     * myCircle.y = -30;
     */
    this.y = y;

    /**
     * The radius of the circle.
     * @type {number}
     * @example
     * // Set the circle radius to 50px.
     * myCircle.radius = 50;
     */
    this.radius = radius;

    /**
     * The opacity in <b>percentage</b>.
     * @type {number}
     * @default
     * @example
     * // Set the opacity to 50%.
     * myCircle.opacity = 50;
     */
    this.opacity = 100;

    /**
     * The colour to fill the circle with.
     * @type {CanvasTexture}
     * @default
     * @example
     * // Set the fill colour to red.
     * myCircle.fillStyle = 'red';
     */
    this.fillStyle = '#ff00ff';

    /**
     * The colour to stroke the circle with.
     * @type {CanvasTexture}
     * @default
     * @example
     * // Set the stroke colour to white.
     * myCircle.strokeStyle = 'white';
     */
    this.strokeStyle = '#ffffff';

    /**
     * Whether to fill the circle.
     * @type {boolean}
     * @default
     * @example
     * // Fill the circle
     * myCircle.fill = true;
     */
    this.fill = true;

    /**
     * Whether to stroke the circle.
     * @type {boolean}
     * @default
     * @example
     * // Stroke the circle
     * myCircle.stroke = true;
     */
    this.stroke = false;

    /**
     * The width of the strokes.
     * @type {number}
     * @default
     * @example
     * // Set the stroke width to 5px.
     * myCircle.lineWidth = 5;
     */
    this.lineWidth = 1;
  }

  /**
   * Used internally to render the circle.
   * @param {CanvasRenderingContext2D} context The renderer's context.
   * @ignore
   */
  render(context) {
    // Set up properties
    context.fillStyle = this.fillStyle;
    context.strokeStyle = this.strokeStyle;
    context.lineWidth = this.lineWidth;
    context.globalAlpha = this.opacity / 100;

    // Draw
    context.beginPath();
    context.arc(this.x, this.y, this.radius, 0, 2 * Math.PI);
    context.closePath();

    // Fill/Stroke
    if (this.fill) {
      context.fill();
    }

    if (this.stroke) {
      context.stroke();
    }
  }
}

module.exports = Circle;
