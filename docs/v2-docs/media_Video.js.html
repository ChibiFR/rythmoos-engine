/**
 * A playable video file.
 * @example
 * import {Video} from 'rythmoos-engine';
 * @example
 * const myVideo = new Video(someVideoElement);
 */
class Video {

  /**
   * Create a Video.
   * @param {HTMLVideoElement} video The video file to use.
   */
  constructor(video) {

    /**
     * The video file to use.
     * @type {HTMLVideoElement}
     * @readonly
     */
    this.file = video;
  }

  /**
   * Whether the video is playing.
   * @type {boolean}
   * @readonly
   */
  get playing() {
    return !this.file.paused;
  }

  /**
   * The duration of the video, in seconds.
   * @type {number}
   * @readonly
   */
  get duration() {
    return this.file.duration;
  }

  /**
   * The current time in the video.
   * @type {number}
   */
  get currentTime() {
    return this.file.currentTime;
  }

  set currentTime(currentTime) {
    this.file.currentTime = currentTime;
  }

  /**
   * The playback rate of the video, in percentage.
   * @type {number}
   */
  get speed() {
    return this.file.playbackRate * 100;
  }

  set speed(speed) {
    this.file.playbackRate = speed / 100;
  }

  /**
   * The volume of the video, in percentage.
   * @type {number}
   */
  get volume() {
    return this.file.volume * 100;
  }

  set volume(volume) {
    this.file.volume = volume / 100;
  }

  /**
   * Whether to loop the video.
   * @type {boolean}
   */
  get loop() {
    return this.file.loop;
  }

  /**
   * Play the video.
   */
  play() {
    this.file.play();
  }

  /**
   * Pause the video.
   */
  pause() {
    this.file.pause();
  }

  /**
   * Stop the video.
   */
  stop() {
    this.file.pause();
    this.file.currentTime = 0;
  }
}

module.exports = Video;
