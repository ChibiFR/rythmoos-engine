/**
 * The root and base class of any game.<br>
 * You can override this class' methods or extend it to create your game.
 * @example <caption>Overriding</caption>
 * const game = new Game(500, 500, document.body);

 * // Set scenes
 * game.scene = new SomeScene(game);

 * // Override method

 * new AwesomeGame(500, 500, document.body);

class Game {

  /**
   * Create a Game.
   * @param {number} width The width of the canvas to create.
   * @param {number} height The height of the canvas to create.
   * @param {HTMLElement} container The parent element which to append the canvas to.
   */
  constructor(width, height, container) {

    /**
     * The game's canvas.
     * @type {HTMLCanvasElement}
     * @readonly
     */
    this.canvas = document.createElement('canvas');

    /**
     * The game's renderer.
     * @type {Renderer}
     * @readonly
     */
    this.renderer = new Renderer(this);

    /**
     * The game's current active scene.
     * @type {Scene}
     * @readonly
     */
    this.scene = new Scene(this);

    // Set up and append the canvas
    this.canvas.width = width;
    this.canvas.height = height;
    container.appendChild(this.canvas);

    // Render an empty scene
    this.renderer.render(this.scene);

    // Start the game
    Loader.waitFor(this.preload).then(() => {
      Loader.waitFor(this.load).then(() => {
        // Initialise the inputs
        Mouse.init(this);

        this.create();

        // The game loop
        const update = (frame) => {
          requestAnimationFrame((frame) => {
            update(frame);
          });

          Time.setFrame(frame);

          this.update();
          this.scene.update();

          for (const name in this.scene.gameObjects) {
            this.scene.gameObjects[name].update();
          }

          this.renderer.render(this.scene);
          
          Mouse.reset();
          Keyboard.reset();
        }

        // Start the loop
        requestAnimationFrame((frame) => {
          update(frame);
        });
      });
    });
  }
