/**
 * The renderer is used to render scenes to the canvas.
 * @example
 * import {Renderer} from 'rythmoos-engine';
 */
class Renderer {

  /**
   * Create a renderer.
   * @param {Game} game A reference to the game instance.
   */
  constructor(game) {

    /**
     * The game's canvas context.
     * @type {CanvasRenderingContext2D}
     * @readonly
     */
    this.context = game.canvas.getContext('2d');

    /**
     * A clone of the context.
     * @type {CanvasRenderingContext2D}
     * @ignore
     */
    this.hiddenContext = game.canvas.cloneNode().getContext('2d');
  }

  /**
   * Render a scene to the canvas.
   * @param {Scene} scene The scene to render.
   */
  render(scene) {
    this.context.fillStyle = '#000000';
    this.context.fillRect(0, 0, this.context.canvas.width, this.context.canvas.height);

    for (const name in scene.gameObjects) {
      const gameObject = scene.gameObjects[name];

      if (!gameObject.visible) continue;

      this.context.save();

      // Translate
      this.context.translate(gameObject.x, gameObject.y);
      
      // Rotate
      if (gameObject.rotation) {
        this.context.translate(gameObject.width / 2, gameObject.height / 2);
        this.context.rotate(gameObject.rotation * Math.PI / 180);
        this.context.translate(-gameObject.width / 2, -gameObject.height / 2);
      }

      // Render
      for (const graphicsName in gameObject.graphics) {
        this.context.save();
        gameObject.graphics[graphicsName].render(this.context);
        this.context.restore();
      }

      this.context.restore();
    }
  }
}

module.exports = Renderer;
