graphics/Image.js

const Graphics = require('./Graphics');

/**
 * A renderable image/video element.
 * @extends Graphics
 * @example
 * import {Image} from 'rythmoos-engine';
 * @example <caption>Creating from an &lt;img&gt; tag</caption>
 * // Get the img element with the class my-img.
 * const myImg = document.querySelector('.my-img');
 * 
 * // Create the image graphics
 * const myImage = new Image(myImg);
 * @example <caption>Creating from loaded files</caption>
 * const myImage = new Image(Loader.get('my image'));
 */
class Image extends Graphics {

  /**
   * Create an Image.
   * @param {HTMLImageElement|HTMLVideoElement} image The image/video element.
   * @param {number} [x=0] The X position (relative to the game object).
   * @param {number} [y=0] The Y position (relative to the game object).
   * @param {number} [width=image.width] The width of the image.
   * @param {number} [height=image.height] The height of the image.
   */
  constructor(image, x = 0, y = 0, width = image.width, height = image.height) {
    super();

    /**
     * The image/video element.
     * @type {HTMLImageElement|HTMLVideoElement}
     * @readonly
     * @example this.image; // => HTMLImageElement or HTMLVideoElement
     */
    this.image = image;

    /**
     * The X position (relative to the game object).
     * @type {number}
     * @example
     * // Move the image 10px to the right
     * this.x += 10;
     */
    this.x = x;

    /**
     * The Y position (relative to the game object).
     * @type {number}
     * @example
     * // Move the image 50px to the bottom
     * this.y += 50;
     */
    this.y = y;

    /**
     * The width. If 0, the image will be rendered to its actual size.
     * @type {number}
     * @example
     * // Change the image width to 248px
     * this.width = 248;
     *
     * // Reset the image's width
     * this.width = 0; // or this.width = this.image.width
     */
    this.width = width;

    /**
     * The height. If 0, the image will be rendered to its actual size.
     * @type {number}
     * @example
     * // Change the image height to 25px
     * this.height = 25;
     * 
     * // Reset the image's height
     * this.height = 0; // or this.height = this.image.height
     */
    this.height = height;

    /**
     * The opacity in percentage.
     * @type {number}
     * @default
     * @example
     * // Making the image half transparent (50%)
     * this.opacity = 50;
     */
    this.opacity = 100;
  }

  /**
   * Used internally to render the image.
   * @param {CanvasRenderingContext2D} context The renderer's context.
   * @ignore
   */
  render(context) {
    const width = this.width || this.image.width;
    const height = this.height || this.image.height;

    context.globalAlpha = this.opacity / 100;
    context.drawImage(this.image, this.x, this.y, width, height);
    
    context.beginPath();
    context.rect(this.x, this.y, this.width, this.height);
    context.closePath();
  }
}

module.exports = Image;
