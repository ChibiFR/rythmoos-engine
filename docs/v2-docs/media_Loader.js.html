/**
 * The loader makes it easy to load and preload assets, before the game starts
 * or on-the-fly.
 * @abstract
 * @example
 * import {Loader} from 'rythmoos-engine';
 * @example <caption>Preloading assets</caption>
 * import {Game, Loader} from 'rythmoos-engine';
 * 
 * // Create a game
 * const game = new Game(500, 500, document.body);
 * 
 * // Load assets before the game starts
 * game.load = () => {
 *   Loader.load('some file', 'assets/some-file.png');
 *   Loader.load('some other file', 'assets/some-other-file.mp3');
 * };
 * @example <caption>Loading assets on-the-fly</caption>
 * // Anywhere outside of the Game.load and Game.preload methods
 * Loader.load('an image', 'an-image.jpg').then(img => {
 *   // Image loaded, also available at Loader.get('an image');
 * });
 * 
 * Loader.load('a video', 'a-video.mp4').then(video => {
 *   // Video loaded, also available at Loader.get('a video');
 * });
 */
class Loader {

  /**
   * Get a loaded file.
   * @param {string} name The name of the file.
   * @return The file or undefined if the file was not loaded.
   * @example
   * // Let's say we loaded a file that we named 'my sprite'
   * Loader.get('my sprite'); // => An HTMLImageElement of our sprite
   * 
   * // Getting a file that we did not load
   * Loader.get('a file that i did not load'); // => undefined
   */
  static get(name) {
    return Loader._files[name];
  }

  /**
   * Load a file.<br>
   * You can tell the Loader which type your file is, or let the Loader guess it
   * from the extension.
   * @param {string} name The name to use for the file.
   * @param {string} pathname The path name to the file to load.
   * @param {'audio'|'video'|'image'|'json'|'other'} [type] The type of file.
   * @return {Promise<*>} A promise that should contain the loaded file.
   * @example
   * // Load images/audios/videos or JSON
   * Loader.load('a file', 'a-file.jpeg').then(myFile => {
   *   // File is loaded, and also available from Loader.get('a file');
   * });
   * 
   * // Or load other types
   * Loader.load('an unknown type of file', 'unknown-type-of.file').then(myFile => {
   *   // File is loaded, and also available from Loader.get('an unknown type of file');
   *   // myFile is a Reponse object from the fetch() method
   * });
   */
  static load(name, pathname) {
    const exp = pathname.split('.');
    const extension = exp[exp.length - 1].toLocaleLowerCase();

    if (['bmp', 'jpg', 'jpeg', 'png', 'gif'].indexOf(extension) !== -1) {
      return Loader.loadImage(name, pathname);
    }
    
    if (['webm', 'mp4', 'avi', 'mov'].indexOf(extension) !== -1) {
      return Loader.loadVideo(name, pathname);
    }
    
    if (['mp3', 'wav', 'aac'].indexOf(extension) !== -1) {
      return Loader.loadAudio(name, pathname);
    }
    
    if (['json'].indexOf(extension) !== -1) {
      return Loader.loadJSON(name, pathname);
    }

    return new Promise((resolve, reject) => {
      const id = Loader._loaded.push(false) - 1;

      fetch(Loader.normalisePath(pathname)).then(response => {
        Loader._files[name] = response;
        Loader._loaded[id] = true;

        resolve(response);
      }).catch(error => {
        console.warn(`Failed to load assets: ${name}`);

        this._loaded[id] = null;

        reject(error);
      });
    });
  }

  /**
   * Load an image.
   * @param {string} name The name to use for the file.
   * @param {string} pathname The pathname to the file.
   * @return A promise that should contain the image element once loaded.
   * @example Loader.loadImage('my image', 'path/to/my/image.png').then(image => {
   *   // Image loaded, also available from Loader.get('my image');
   * });
   */
  static loadImage(name, pathname) {
    return new Promise((resolve, reject) => {
      const image = document.createElement('img');
      const id = Loader._loaded.push(false) - 1;

      image.onload = () => {
        Loader._files[name] = image;
        Loader._loaded[id] = true;

        resolve(image);
      };

      image.onerror = (err) => {
        console.warn(`Failed to load asset: ${name}`);

        this._loaded[id] = null;

        reject(err);
      };

      image.src = Loader.normalisePath(pathname);
    });
  }

  /**
   * Load a video.
   * @param {string} name The name to use for the file.
   * @param {string} pathname The pathname to the file.
   * @return A promise that should contain the video element once loaded.
   * @example Loader.loadVideo('my video', 'path/to/my/video.mp4').then(video => {
   *   // Video loaded, also available from Loader.get('my video');
   * });
   */
  static loadVideo(name, pathname) {
    return new Promise((resolve, reject) => {
      const video = document.createElement('video');
      const id = Loader._loaded.push(false) - 1;

      video.onloadeddata = () => {
        Loader._files[name] = video;
        Loader._loaded[id] = true;

        resolve(video);
      };

      video.onerror = (err) => {
        console.warn(`Failed to load asset: ${name}`);

        this._loaded[id] = null;

        reject(err);
      };

      video.src = Loader.normalisePath(pathname);
    });
  }

  /**
   * Load an audio.
   * @param {string} name The name to use for the file.
   * @param {string} pathname The pathname to the file.
   * @return A promise that should contain the audio element once loaded.
   * @example Loader.loadAudio('my audio', 'path/to/my/audio.mp3').then(audio => {
   *   // Audio loaded, also available from Loader.get('my audio');
   * });
   */
  static loadAudio(name, pathname) {
    return new Promise((resolve, reject) => {
      const audio = document.createElement('audio');
      const id = Loader._loaded.push(false) - 1;

      audio.onloadeddata = () => {
        Loader._files[name] = audio;
        Loader._loaded[id] = true;

        resolve(audio);
      };

      audio.onerror = (err) => {
        console.warn(`Failed to load asset: ${name}`);

        this._loaded[id] = null;

        reject(err);
      };

      audio.src = Loader.normalisePath(pathname);
    });
  }

  /**
   * Load JSON data.
   * @param {string} name The name to use for the file.
   * @param {string} pathname The pathname to the JSON file.
   * @return A promise that should contain the parsed JSON once loaded.
   * @example Loader.loadJSON('my json', 'path/to/my/json.json').then(json => {
   *   // JSON loaded, also available from Loader.get('my json');
   * });
   */
  static loadJSON(name, pathname) {
    return new Promise((resolve, reject) => {
      const id = Loader._loaded.push(false) - 1;

      fetch(Loader.normalisePath(pathname)).then(response => {
        response.json().then(value => {
          this._files[name] = value;
          this._loaded[id] = true;

          resolve(value);
        }).catch(error => {
          console.warn(`Failed to parse JSON asset: ${name}`);

          this._loaded[id] = null;

          reject(error);
        });
      }).catch(error => {
        console.warn(`Failed to load asset: ${name}`);

        reject(error);
      });
    });
  }

  /**
   * Used internally to run a function that loads file then return a promise.
   * @param {function} callback The function to call and wait for.
   * @return {Promise<void>} An empty promise once the files are loaded.
   * @ignore
   */
  static waitFor(callback) {
    return new Promise((resolve, reject) => {
      callback();

      if (Loader._loaded.indexOf(false) === -1) {
        resolve();
      }

      let interval = setInterval(() => {
        if (Loader._loaded.indexOf(false) === -1) {
          clearInterval(interval);

          resolve();
        }

        if (Loader._loaded.indexOf(null) !== -1) {
          clearInterval(interval);

          reject();
        }
      }, 0);
    });
  }

  /**
   * Normalise a pathname with the Loader's basePath.
   * @param {string} pathname The pathname to normalise.
   * @return {string} The normalised path.
   * @private
   */
  static normalisePath(pathname) {
    Loader._a.href = `${Loader.basePath}/${pathname}`;

    return Loader._a.href;
  }
}

/**
 * The base path to use for loading assets.
 * @type {string}
 * @default
 * @example
 * // Let's say all our assets are in the assets folder, we dont want to specify
 * // the assets directory in the path everytime we load an assets
 * Loader.basePath = 'assets';
 * 
 * // We can now load, for example, `assets/my-image.png` as so:
 * Loader.load('my image', 'my-image.png');
 */
Loader.basePath = '.';

/**
 * The loaded files.
 * @type {object}
 * @private
 */
Loader._files = {};

/**
 * The amount of files loaded.
 * @type {boolean[]}
 * @private
 */
Loader._loaded = [];

/**
 * Used to normalise paths.
 * @type {HTMLElement}
 * @private
 */
Loader._a = document.createElement('a');

module.exports = Loader;
