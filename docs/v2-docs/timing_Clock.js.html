const ClockMode = require('./ClockMode');

/**
 * The Clock provides helpers for working with the user's system date and time.
 * @example
 * import {Clock} from 'rythmoos-engine';
 */
class Clock {
  
  /**
   * The current UTC time's timestamp, in milliseconds.
   * @type {number}
   * @readonly
   * @example
   * // Get the current UTC time.
   * Clock.time;
   */
  static get time() {
    return new Date().getTime();
  }

  /**
   * The current time's hours.
   * @type {number}
   * @readonly
   * @example
   * // Let's say it's 3pm
   * Clock.hours; // => 3
   */
  static get hours() {
    return new Date().getHours();
  }

  /**
   * The current time's minutes.
   * @type {number}
   * @readonly
   * @example
   * // Let's say it's 3:04pm
   * Clock.minutes; // => 4
   */
  static get minutes() {
    return new Date().getMinutes();
  }

  /**
   * The current time's seconds.
   * @type {number}
   * @readonly
   * @example
   * // Let's say it's 3:04:54pm
   * Clock.seconds; // => 54
   */
  static get seconds() {
    return new Date().getSeconds();
  }

  /**
   * The current time's milliseconds.
   * @type {number}
   * @readonly
   * @example
   * // Let's say it's 15 hours, 4 minutes, 54seconds and 542ms
   * Clock.milliseconds; // => 542
   */
  static get milliseconds() {
    return new Date().getMilliseconds();
  }

  /**
   * The current day of the week, from 0 (sunday) to 6 (saturday).
   * @type {number}
   * @readonly
   * @example
   * // Let's say it's friday
   * Clock.day; // => 5
   */
  static get day() {
    return new Date().getDay();
  }

  /**
   * The current day of the month.
   * @type {number}
   * @readonly
   * @example
   * // Let's say it's November, 14th
   * Clock.date; // => 14
   */
  static get date() {
    return new Date().getDate();
  }

  /**
   * The current month.
   * @type {number}
   * @readonly
   * @example
   * // Let's say we're in November
   * Clock.month; // => 11
   */
  static get month() {
    return new Date().getMonth();
  }

  /**
   * The current day as a 3 letter lower cased string (e.g Sunday will be sun).
   * @type {string}
   * @readonly
   * @example
   * // Let's say it's friday
   * Clock.day3; // => 'fri'
   */
  static get day3() {
    return new Date().toString().toLowerCase().substr(0, 3);
  }

  /**
   * The current month as a 3 letter lower cased string (e.g November will be nov).
   * @type {string}
   * @readonly
   * @example
   * // Let's say we're in November
   * Clock.month3; // => 'nov'
   */
  static get month3() {
    return new Date().toString().toLowerCase().substr(4, 3);
  }

  /**
   * The current clock's time.
   * @type {string}
   * @readonly
   * @example
   * // At 3:04:47pm
   * Clock.now; // => '15:04:47' (with Clock.clockMode set to H24)
   * // or
   * Clock.now; // => '3:04:47pm' (with Click.clockMode set to H12)
   */
  static get now() {
    const date = new Date();
    let h = date.getHours().toString();
    let m = date.getMinutes().toString();
    let s = date.getSeconds().toString();
    let now = '';

    // Normalise
    if (this.clockMode === ClockMode.H24) {
      h = `00${h}`.substr(h.length, 2);
    }

    m = `00${m}`.substr(m.length, 2);
    s = `00${s}`.substr(s.length, 2);

    // Format
    if (this.clockMode === ClockMode.H24) {
      now = `${h}:${m}:${s}`;
    } else {
      if (parseInt(h) < 11) {
        now = `${h}:${m}:${s}am`;
      } else {
        if (parseInt(h) > 12) {
          h = parseInt(h) - 12;
        }

        now = `${h}:${m}:${s}pm`;
      }
    }

    return now;
  }
}

/**
 * The {@link ClockMode|mode} to use for the clock.
 * @type {ClockMode}
 * @default
 * @example
 * import {Clock, ClockMode} from 'rythmoos-engine';
 * 
 * // 24-hour format
 * Clock.clockMode = ClockMode.H24;
 * console.log(Clock.now); // => '15:04:47'
 * 
 * // 12-hour-format
 * Clock.clockMode = ClockMode.H12;
 * console.log(Clock.now); // => '3:04:47pm'
 */
Clock.clockMode = ClockMode.H24;

module.exports = Clock;
