const MouseButton = require('./MouseButton');

/**
 * The mouse input.<br>
 * It contains all information about the cursor and mouse buttons and wheel.
 * @abstract
 * @example
 * import {Mouse} from 'rythmoos-engine';
 */
class Mouse {

  /**
   * Used internally to initialise the mouse input.
   * @param {Game} game The game instance.
   * @ignore
   */
  static init(game) {
    const canvas = game.canvas;

    for (const button in MouseButton) {
      Mouse._buttons[MouseButton[button]] = false;
      Mouse._clicks[MouseButton[button]] = false;
    }

    window.addEventListener('mousemove', e => {
      Mouse.cursorDetected = true;
      Mouse.cursorX = e.clientX - canvas.offsetLeft;
      Mouse.cursorY = e.clientY - canvas.offsetTop;
      Mouse.deltaX += e.movementX;
      Mouse.deltaY += e.movementY;

      if (Mouse.cursorX < 0 || Mouse.cursorX > canvas.width ||
          Mouse.cursorY < 0 || Mouse.cursorY > canvas.height) {
        for (const button in MouseButton) {
          Mouse._buttons[MouseButton[button]] = false;
        }
      }
    });

    canvas.addEventListener('mousedown', e => {
      if (Mouse.preventDefault) e.preventDefault();

      Mouse._buttons[e.button] = true;
      Mouse._clicks[e.button] = true;
      Mouse.lastKey = e.button;
    });

    canvas.addEventListener('mouseup', e => {
      if (!Mouse._buttons[e.button]) return;
      if (Mouse.preventDefault) e.preventDefault();

      Mouse._buttons[e.button] = false;
    });

    window.addEventListener('wheel', e => {
      if (Mouse.preventDefault) e.preventDefault();

      Mouse.scrollX += e.deltaX;
      Mouse.scrollY += e.deltaY;
    });

    canvas.addEventListener('contextmenu', e => {
      if (Mouse.preventDefault) e.preventDefault();
    });

    window.addEventListener('blur', () => {
      for (const button in Mouse._buttons) {
        Mouse._buttons[button] = false;
      }
    });
  }

  /**
   * Used internally to reset the input states.
   * @ignore
   */
  static reset() {
    Mouse.deltaX = 0;
    Mouse.deltaY = 0;
    Mouse.scrollX = 0;
    Mouse.scrollY = 0;
    Mouse.lastKey = null;

    for (const button in Mouse._clicks) {
      Mouse._clicks[button] = false;
    }
  }

  /**
   * Whether the specified button(s) is/are down.
   * @param {...MouseButton} buttons The button(s) to check.
   * @return {boolean} Whether the button(s) is/are down.
   * @example
   * import {Mouse, MouseButton} from 'rythmoos-engine';
   * 
   * // ...
   * 
   * if (Mouse.buttonDown(MouseButton.LEFT)) {
   *   // The mouse left button is down
   * }
   */
  static buttonDown(...buttons) {
    if (buttons.length === 0) return false;

    for (const button of buttons) {
      if (!Mouse._buttons[button]) {
        return false;
      }
    }

    return true;
  }

  /**
   * Whether the specified button(s) was/were clicked.
   * @param {...MouseButton} buttons The button(s) to check.
   * @return {boolean} Whether the button(s) was/were clicked.
   * @example
   * import {Mouse, MouseButton} from 'rythmoos-engine';
   *
   * // ...
   *
   * if (Mouse.buttonClicked(MouseButton.LEFT)) {
   *   // The mouse left button was clicked
   * }
   */
  static buttonClicked(...buttons) {
    if (buttons.length === 0) return false;

    for (const button of buttons) {
      if (!Mouse._clicks[button]) {
        return false;
      }
    }

    return true;
  }
}

/**
 * The button states.
 * @type {object}
 * @private
 */
Mouse._buttons = {};

/**
 * The click states.
 * @type {object}
 * @static
 * @private
 */
Mouse._clicks = {};

/**
 * The cursor position in the X axis.
 * @type {number}
 * @readonly
 * @example Mouse.cursorX;
 */
Mouse.cursorX = 0;

/**
 * The cursor position in the Y axis.
 * @type {number}
 * @readonly
 * @example Mouse.cursorY;
 */
Mouse.cursorY = 0;

/**
 * The cursor movement in the X axis (in pixels), since the last frame.
 * @type {number}
 * @readonly
 * @example Mouse.deltaX;
 */
Mouse.deltaX = 0;

/**
 * The cursor movement in the Y axis (in pixels), since the last frame.
 * @type {number}
 * @readonly
 * @example Mouse.deltaY;
 */
Mouse.deltaY = 0;

/**
 * The scroll amount in the X axis (in pixels), since the last frame.
 * @type {number}
 * @readonly
 * @example Mouse.scrollX;
 */
Mouse.scrollX = 0;

/**
 * The scroll amount in the Y axis (in pixels), since the last frame.
 * @type {number}
 * @readonly
 * @example Mouse.scrollY;
 */
Mouse.scrollY = 0;

/**
 * The last button clicked since the last frame, null if no button was clicked.<br>
 * Similar to {@link Keyboard#lastKey|Keyboard.lastKey} but with mouse buttons.
 * @type {MouseButton|null}
 * @readonly
 * @example Mouse.lastKey;
 */
Mouse.lastKey = null;

/**
 * Whether the cursor was detected.<br>
 * This will be true once the cursor position is detected (first mousemove event).
 * @type {boolean}
 * @readonly
 * @example
 * if (Mouse.cursorDetected) {
 *   // Cursor was detected
 * } else {
 *   // Cursor was not detected, Mouse.cursorX and Mouse.cursorY still have their default
 *   // values (0, 0).
 * }
 */
Mouse.cursorDetected = false;

/**
 * Whether to prevent the mouse events' default actions.
 * @type {boolean}
 * @default
 * @example
 * Mouse.preventDefault = true; // Native mouse events are disabled
 * Mouse.preventDefault = false; // Native mouse events are enabled
 */
Mouse.preventDefault = false;

module.exports = Mouse;
