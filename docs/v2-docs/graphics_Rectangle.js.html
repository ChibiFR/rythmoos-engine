graphics/Rectangle.js

const Graphics = require('./Graphics');

/**
 * A simple rectangle with possible rounded corners.
 * @extends Graphics
 * @example
 * import {Rectangle} from 'rythmoos-engine';
 * @example <caption>Creating a Rectangle</caption>
 * const myRect = new Rectangle();
 */
class Rectangle extends Graphics {

  /**
   * Create a Rectangle.
   * @param {number} [x=0] The X position (relative to the game object).
   * @param {number} [y=0] The Y position (relative to the game object).
   * @param {number} [width=180] The width of the rectangle.
   * @param {number} [height=100] The height of the rectangle.
   * @param {number} [radius=0] The radius of the rectangle's borders.
   */
  constructor(x = 0, y = 0, width = 180, height = 100, radius = 0) {
    super();

    /**
     * The X position (relative to the game object).
     * @type {number}
     */
    this.x = x;

    /**
     * The Y position (relative to the game object).
     * @type {number}
     */
    this.y = y;

    /**
     * The width of the rectangle.
     * @type {number}
     */
    this.width = width;

    /**
     * The height of the rectangle.
     * @type {number}
     */
    this.height = height;

    /**
     * The rectangle's borders radius.
     * @type {number}
     */
    this.radius = radius;
    
    /**
     * The opacity in percentage.
     * @type {number}
     * @default
     */
    this.opacity = 100;

    /**
     * The colour to fill the rectangle with.
     * @type {CanvasTexture}
     * @default
     */
    this.fillStyle = '#ff00ff';

    /**
     * The colour to stroke the rectangle with.
     * @type {CanvasTexture}
     * @default
     */
    this.strokeStyle = '#ffffff';

    /**
     * Whether to fill the rectangle.
     * @type {boolean}
     * @default
     */
    this.fill = true;

    /**
     * Whether to stroke the rectangle.
     * @type {boolean}
     * @default
     */
    this.stroke = false;

    /**
     * The width of the strokes.
     * @type {number}
     * @default
     */
    this.lineWidth = 1;
  }

  /**
   * Used internally to render the rectangle.
   * @param {CanvasRenderingContext2D} context The renderer's context.
   * @ignore
   */
  render(context) {
    // Set up properties
    context.fillStyle = this.fillStyle;
    context.strokeStyle = this.strokeStyle;
    context.lineWidth = this.lineWidth;
    context.globalAlpha = this.opacity / 100;

    // Draw
    context.beginPath();

    if (this.radius) {
      context.moveTo(this.x + this.radius, this.y);
      context.lineTo(this.x + this.width - this.radius, this.y);
      context.arcTo(
        this.x + this.width, this.y,
        this.x + this.width, this.y + this.radius,
        this.radius
      );
      context.lineTo(this.x + this.width, this.y + this.height - this.radius);
      context.arcTo(
        this.x + this.width, this.y + this.height,
        this.x + this.width - this.radius, this.y + this.height,
        this.radius
      );
      context.lineTo(this.x + this.radius, this.y + this.height);
      context.arcTo(
        this.x, this.y + this.height,
        this.x, this.y - this.radius,
        this.radius
      );
      context.lineTo(this.x, this.y + this.radius);
      context.arcTo(
        this.x, this.y,
        this.x + this.radius, this.y,
        this.radius
      );
    } else {
      context.rect(this.x, this.y, this.width, this.height);
    }

    context.closePath();

    // Fill/Stroke
    if (this.fill) {
      context.fill();
    }

    if (this.stroke) {
      context.stroke();
    }
  }
}

module.exports = Rectangle;
