/**
 * A game object is a game entity, usually containing graphics (e.g player
 * character).
 */
class GameObject {

  /**
   * Create a GameObject.
   * @param {Scene} scene A reference to the parent scene.
   * @param {number} [x=0] The X position.
   * @param {number} [y=0] The Y position.
   * @param {number} [width=0] The width.
   * @param {number} [height=0] The height.
   */
  constructor(scene, x = 0, y = 0, width = 0, height = 0) {

    /**
     * A reference to the parent scene.
     * @type {Scene}
     * @readonly
     */
    this.scene = scene;

    /**
     * A reference to the game instance.
     * @type {Game}
     * @readonly
     */
    this.game = this.scene.game;

    /**
     * The graphics contained in this game object.
     * @type {Map<Graphics>}
     * @readonly
     */
    this.graphics = {};

    /**
     * The X position.
     * @type {number}
     */
    this.x = x;

    /**
     * The Y position.
     * @type {number}
     */
    this.y = y;

    /**
     * The width of the game object.
     * @type {number}
     */
    this.width = width;

    /**
     * The height of the game object.
     * @type {number}
     */
    this.height = height;

    /**
     * The rotation in degrees.
     * @type {number}
     * @default
     */
    this.rotation = 0;

    /**
     * Whether to render the game object.
     * @type {boolean}
     * @default
     */
    this.visible = true;

    this.create();
  }

   * Run when the game object is created.

   * import {GameObject, Time} from 'rythmoos-engine';

   *   update() {

   *     this.x += 30 * Time.second;
   *   }
   * }

  /**
   * Get a graphics from the game object.
   * @param {string} key The key of the graphics.
   * @return {Graphics} The graphics, or undefined if it was not defined.
   * @example
   * this.get('my rectangle').x = 45;
   * this.get('my text').text = 'qwerty';
   */
  get(key) {
    return this.graphics[key];
  }

  /**
   * Set (add/update) a graphics in the game object.
   * @param {string} key The key of the graphics.
   * @param {Graphics} graphics The graphics.
   * @example
   * this.set('my rectangle', new Rectangle(20, 20, 20, 20));
   * this.set('my text', new Text('some text', 0, 0, Font.default));
   */
  set(key, graphics) {
    this.graphics[key] = graphics;
  }

  /**
   * Remove a graphics from the scene.
   * @param {string} key The key of the graphics.
   * @example
   * this.remove('my rectangle');
   * this.remove('my text');
   */
  remove(key) {
    delete this.graphics[key];
  }

  /**
   * Check if a point (x, y) is in the game object.
   * @param {number} x The X position of the point.
   * @param {number} y The Y position of the point.
   * @return {boolean} Whether the point is in the game object. This is based on the graphics,
   * not the game object's x, y, width and height properties.
   * @example <caption>Checking for mouse hovers</caption>
   * if (this.contains(Mouse.cursorX, Mouse.cursorY)) {
   *   // Mouse is hovering the game object.
   * }
   */
  contains(x, y) {
    const context = this.game.renderer.hiddenContext;

    for (const graphics in this.graphics) {
      context.save();

      context.translate(this.x, this.y);

      if (this.rotation) {
        context.translate(this.width / 2, this.height / 2);
        context.rotate(this.rotation * Math.PI / 180);
        context.translate(-this.width / 2, -this.height / 2)
      }

      this.graphics[graphics].render(context);

      context.restore();

      if (context.isPointInPath(x, y)) {
        return true;
      }
    }

    return false;
  }
}

module.exports = GameObject;
