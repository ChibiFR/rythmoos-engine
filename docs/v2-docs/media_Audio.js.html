media/Audio.js

/**
 * A <b>playable audio</b> file that can be analysed through its <b>FFT data</b>.<br>
 * The instance <i>inherits</i> from the {@link Audio#file|audio file} passed in the
 * constructor (loop, volume, etc).
 * @example
 * import {Audio} from 'rythmoos-engine';
 * @example <caption>Creating from an &lt;audio&gt; tag</caption>
 * const myAudioTag = document.querySelector('.my-audio-tag');
 * const myAudio = new Audio(myAudioTag);
 * @example <caption>Creating from a loaded audio</caption>
 * const myAudio = new Audio(Loader.get('my-audio'));
 */
class Audio {

  /**
   * Create an audio.
   * @param {HTMLAudioElement} audio The audio file to use.
   * @param {number} [fftSize=2048] The size of the fft buffer.
   */
  constructor(audio, fftSize = 2048) {

    /**
     * The audio file to use.
     * @type {HTMLAudioElement}
     * @readonly
     * @example myAudio.file; // HTMLAudioElement
     */
    this.file = audio;

    /**
     * The audio context.
     * @type {AudioContext}
     * @private
     */
    this._audioContext = new AudioContext();

    /**
     * The media element source.
     * @type {MediaElementAudioSourceNode}
     * @private
     */
    this._source = this._audioContext.createMediaElementSource(this.file);

    /**
     * The audio analyser.
     * @type {AnalyserNode}
     * @private
     */
    this._analyser = this._audioContext.createAnalyser();

    this._analyser.fftSize = fftSize;

    /**
     * The fft frequency buffer.
     * @type {Uint8Array}
     * @private
     */
    this._fftFrequencyBuffer = new Uint8Array(this._analyser.frequencyBinCount);

    /**
     * The fft waveform buffer.
     * @type {Uint8Array}
     * @private
     */
    this._fftWaveformBuffer = new Uint8Array(this._analyser.frequencyBinCount);
    
    this._source.connect(this._analyser);
    this._analyser.connect(this._audioContext.destination);
  }

  /**
   * Whether the audio is playing.
   * @type {boolean}
   * @readonly
   * @example
   * // Create an audio
   * const myAudio = new Audio(someAudioElement);
   * 
   * myAudio.play();
   * console.log(myAudio.playing); // => true
   * 
   * myAudio.stop();
   * console.log(myAudio.playing); // => false
   */
  get playing() {
    return !this.file.paused;
  }

  /**
   * The duration in <i>seconds</i> of the audio file.
   * @type {number}
   * @readonly
   * @example
   * // Assume we loaded a 3 minutes long audio file named 'my audio'.
   * const myAudio = new Audio(Loader.get('my audio'));
   * 
   * console.log(myAudio.duration); // => // 180
   */
  get duration() {
    return this.file.duration;
  }

  /**
   * The current time, in <i>seconds</i>, in the track.
   * @type {number}
   * @example <caption>Getting the current time</caption>
   * // Assume our myAudio object has been playing for 15 seconds.
   * console.log(myAudio.currentTime); // => 15
   * @example <caption>Setting the current time</caption>
   * myAudio.currentTime = 60; // Skips to 60 sec
   * console.log(myAudio.currentTime); // => 60
   * 
   * myAudio.currentTime -= 30; // Skips back to 30 sec
   * console.log(myAudio.currentTime); // => 30
   */
  get currentTime() {
    return this.file.currentTime;
  }

  set currentTime(currentTime) {
    this.file.currentTime = currentTime;
  }

  /**
   * The playback speed of the audio, in <b>percentage</b>.
   * @type {number}
   * @example
   * console.log(myAudio.speed); // => 100
   * 
   * // Play the audio at 2x speed
   * myAudio.speed = 200; // 200% speed
   * 
   * // Go back to normal speed
   * myAudio.speed = 100; // 100% speed
   *
   */
  get speed() {
    return this.file.playbackRate * 100;
  }

  set speed(speed) {
    this.file.playbackRate = speed / 100;
  }

  /**
   * The volume of the audio, in <b>percentage</b>.
   * @type {number}
   * @example
   * console.log(myAudio.volume); // => 100
   * 
   * // Play the audio at half volume
   * myAudio.volume = 50; // 50%
   * 
   * // Play the audio at max volume
   * myAudio.volume = 100; // 100%
   */
  get volume() {
    return this.file.volume * 100;
  }

  set volume(volume) {
    this.file.volume = volume / 100;
  }

  /**
   * Whether to loop the audio.
   * @type {boolean}
   * @example
   * console.log(myAudio.loop) // => false
   * 
   * // Make the audio loop (auto restarts when it ends)
   * myAudio.loop = true;
   * 
   * console.log(myAudio.loop); // => true
   */
  get loop() {
    return this.file.loop;
  }

  set loop(loop) {
    this.file.loop = loop;
  }

  /**
   * The size of the fft buffer for <i>frequency and waveform</i> data analysis.<br>
   * Must be a <b>non-zero power of 2 between 32 and 32768</b>.
   * @type {number}
   * @example
   * myAudio.fftSize = 2048;
   * console.log(myAudio.getFrequency().length); // => 2048
   * 
   * myAudio.fftSize = 512;
   * console.log(myAudio.getFrequency().length); // => 512
   */
  get fftSize() {
    return this._analyser.frequencyBinCount;
  }

  set fftSize(fftSize) {
    this._analyser.fftSize = fftSize;
    this._fftFrequencyBuffer = new Uint8Array(this.fftSize);
    this._fftWaveformBuffer = new Uint8Array(this.fftSize);
  }

  /**
   * Play the audio.
   * @example myAudio.play();
   */
  play() {
    this.file.play();
  }

  /**
   * Pause the audio.
   * @example myAudio.pause();
   */
  pause() {
    this.file.pause();
  }

  /**
   * Stop the audio.
   * @example
   * console.log(myAudio.currentTime); // => 56
   * 
   * myAudio.stop();
   * console.log(myAudio.currentTime); // => 0
   */
  stop() {
    this.file.pause();
    this.file.currentTime = 0;
  }

  /**
   * Get the <i>byte</i> frequency data.<br>
   * The size of the array is defined by the {@link Audio#fftSize|fftSize} property.
   * @return {Uint8Array}
   * @example myAudio.getFrequency();
   */
  getFrequency() {
    this._analyser.getByteFrequencyData(this._fftFrequencyBuffer);

    return this._fftFrequencyBuffer;
  }

  /**
   * Get the <i>byte</i> waveform data.<br>
   * The size of the array is defined by the {@link Audio#fftSize|fftSize} property.
   * @return {Uint8Array}
   * @example myAudio.getWaveform();
   */
  getWaveform() {
    this._analyser.getByteTimeDomainData(this._fftWaveformBuffer);

    return this._fftWaveformBuffer;
  }
}

module.exports = Audio;
