/**
 * A sound is a short audio file that can be played more than once at the
 * same time.
 */
class Sound {

  /**
   * Create a Sound.
   * @param {HTMLAudioElement} file The audio file to use.
   */
  constructor(file) {

    /**
     * The audio file.
     * @type {HTMLAudioElement}
     */
    this.file = file;

    /**
     * The tracks.
     * @type {HTMLAudioElement[]}
     * @private
     */
    this._tracks = [this.file];

    for (let i = 1; i < Sound.maxNodes; i++) {
      this._tracks.push(this.file.cloneNode());
    }
  }

  /**
   * Whether a node of this sound is playing.
   * @type {boolean}
   * @readonly
   */
  get playing() {
    for (const track of this._tracks) {
      if (!track.paused) {
        return true;
      }
    }

    return false;
  }

  /**
   * The sound's volume in pergentage. Defaults to 100.
   * @type {number}
   */
  get volume() {
    return this.file.volume * 100;
  }

  set volume(volume) {
    for (const track of this._tracks) {
      track.volume = volume / 100;
    }
  }

  /**
   * Play the sound.
   * @return The ID of the node, or null if no node is available.
   */
  play() {
    for (let i = 0; i < this._tracks.length; i++) {
      if (this._tracks[i].paused) {
        this._tracks[i].play();

        return i;
      }
    }

    return null;
  }

  /**
   * Stop the sound, or a specific node.
   * @param {number} [id] The id of the node to stop.
   */
  stop(id) {
    if (id === undefined) {
      for (const track of this._tracks) {
        track.pause();
        track.currentTime = 0;
      }
    } else {
      this._tracks[id].pause();
      this._tracks[id].currentTime = 0;
    }
  }
}

/**
 * The maximum nodes that can be played at the same time.
 * @type {number}
 * @readonly
 */
Sound.maxNodes = 3;

module.exports = Sound;
