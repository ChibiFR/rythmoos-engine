graphics/Text.js

const Graphics = require('./Graphics');
const Font = require('../utils/Font');

/**
 * A text.
 */
class Text extends Graphics {
  constructor(text = '', x = 0, y = 0, font = Font.default) {
    super();

    /**
     * The context of the text.
     * @type {string}
     */
    this.text = text;

    /**
     * The X position (relative to the game object).
     * @type {number}
     */
    this.x = x;

    /**
     * The Y position (relative to the game object).
     * @type {number}
     */
    this.y = y;

    /**
     * The font to use for the text.
     * @type {Font}
     */
    this.font = font;

    /**
     * The text alignment.<br>
     * This is used to position the text from its left, right, or any valid text
     * alignment.
     * @type {TextAlignment}
     * @default
     */
    this.align = 'left';

    /**
     * The text's baseline.<br>
     * This is used to position the text from its top, bottom, or any valid text
     * baseline.
     * @type {TextBaseline}
     * @default
     */
    this.baseline = 'top';

    /**
     * The colour to fill the text with.
     * @type {CanvasTexture}
     * @default
     */
    this.fillStyle = '#ffffff';

    /**
     * The colour to stroke the text with.
     * @type {CanvasTexture}
     * @default
     */
    this.strokeStyle = '#444444';

    /**
     * Whether to fill the text.
     * @type {boolean}
     * @default
     */
    this.fill = true;

    /**
     * Whether to stroke the text.
     * @type {boolean}
     * @default
     */
    this.stroke = false;

    /**
     * The opacity in percent.
     * @type {number}
     * @default
     */
    this.opacity = 100;
  }

  /**
   * Used internally to render the text.
   * @param {CanvasRenderingContext2D} context The renderer's context.
   * @ignore
   */
  render(context) {
    // Set up properties
    context.font = this.font.toString();
    context.textAlign = this.align;
    context.textBaseline = this.baseline;
    context.fillStyle = this.fillStyle;
    context.strokeStyle = this.strokeStyle;
    
    // Fill/Stroke
    if (this.fill) {
      context.fillText(this.text, this.x, this.y);
    }

    if (this.stroke) {
      context.strokeText(this.text, this.x, this.y);
    }
  }
}

module.exports = Text;
