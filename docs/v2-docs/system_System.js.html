const OS = require('./OS');

/**
 * Contain information about the user's system and browser.
 * @abstract
 * @example
 * import {System} from 'rythmoos-engine';
 */
class System {

  /**
   * Whether the cookies are enabled on the user's web browser.
   * @type {boolean}
   * @readonly
   * @example
   * if (System.cookieEnabled) {
   *   // You can create cookies
   * }
   */
  static get cookieEnabled() {
    return navigator.cookieEnabled;
  }

  /**
   * The number of physical CPU cores of the user's machine.
   * @type {number}
   * @readonly
   * @example
   * // On a 4 cored CPU PC:
   * console.log(System.cpus); // => 4
   * 
   * // On a 6 cored CPU PC:
   * console.log(System.cpus); // => 6
   */
  static get cpus() {
    return navigator.hardwareConcurrency;
  }

  /**
   * The language of the user (e.g 'en').
   * @type {string}
   * @readonly
   * @example
   * // Detected if the user uses English
   * if (System.language.substr(0, 2)) { // could be 'en' or 'en-US', etc... so, substr(0, 2)
   *   // User uses English
   * }
   */
  static get language() {
    return navigator.language;
  }

  /**
   * The user's operating system.
   * @type {OS}
   * @readonly
   * @example
   * switch (System.os) {
   *   case OS.WINDOWS:
   *     console.log('You are using Window').
   *     break;
   *   case OS.MAC:
   *     console.log('You are using Mac OS').
   *     break;
   *   case OS.LINUX:
   *   default:
   *     // If not found, will default to Linux
   *     console.log('You are not using Windows/Mac');
   * }
   */
  static get os() {
    const appVersion = navigator.appVersion.toLocaleLowerCase();

    if (appVersion.indexOf('win') !== -1) {
      return OS.WINDOWS;
    } else if (appVersion.indexOf('mac') !== -1) {
      return OS.MAC;
    } else {
      return OS.LINUX;
    }
  }
}

module.exports = System;
