timing/Timer.js

const Time = require('./Time');

/**
 * The Timer class is an easy way to set up a countdown in your game.
 * @example
 * import {Timer} from 'rythmoos-engine';
 * @example
 * const myTimer = new Timer();
 */
class Timer {

  /**
   * Create a Timer.
   * @param {number} duration The duration of the timer, in milliseconds.
   */
  constructor(duration) {

    /**
     * The duration of the timer.
     * @type {number}
     * @example myTimer.duration = 5 * 1000; // 5 * 1000ms = 5s
     */
    this.duration = duration;

    /**
     * The time when the timer was started.
     * @type {number}
     * @private
     */
    this._startedAt = 0;
  }

  /**
   * Whether the timer is complete.
   * @type {boolean}
   * @readonly
   * @example
   * // Create a 20 sec timer
   * const myTimer = new Timer(20000);
   * 
   * // In some game object's create method
   * myGameObject.create = () => {
   *   myTimer.start();
   * }
   * 
   * // Check whether it's done, in some game object's update method
   * myGameObject.update = () => {
   *   if (myTimer.done) {
   *     console.log('20 sec have elapsed');
   *   }
   * }
   */
  get done() {
    return Time.time > this._startedAt + this.duration;
  }

  /**
   * The time elapsed since the timer was started, equals to the duration once it
   * is complete.
   * @type {number}
   * @readonly
   * @example
   * const myTimer = new Timer(5000);
   * 
   * myTimer.start();
   * 
   * setTimeout(() => {
   *   console.log(myTimer.elapsed); // appr. 1000ms
   * }, 1000);
   */
  get elapsed() {
    return (this.done ? this.duration : Time.time - this._startedAt);
  }

  /**
   * Start or restart the timer.
   * @example
   * const myTimer = new Timer(1000);
   * myTimer.start();
   */
  start() {
    this._startedAt = Time.time;
  }
}

module.exports = Timer;
