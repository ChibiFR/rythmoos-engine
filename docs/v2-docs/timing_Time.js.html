timing/Time.js

/**
 * Contain all information about time.
 * @abstract
 * @example
 * import {Time} from 'rythmoos-engine';
 */
class Time {

  /**
   * Used internally to update the time information.
   * @param {number} frame The timestamp passed in by the browser.
   * @ignore
   */
  static setFrame(frame) {
    Time._deltaTime = frame - this._time;
    Time._time = frame;
  }

  /**
   * The time of the animation frame, provided by the browser, in milliseconds.
   * @type {number}
   * @readonly
   */
  static get time() {
    return Time._time;
  }

  /**
   * The time elapsed since the last frame, in milliseconds.
   * @type {number}
   * @readonly
   * @example <caption>Creating animations with deltaTime</caption>
   * // In some game object's update method...
   * 
   * // Move the game object to the right, at the same speed, whatever the frame rate is
   * this.x += 0.1 * Time.deltaTime;
   */
  static get deltaTime() {
    return Time._deltaTime;
  }

  /**
   * The amount of frames per second, based on the current delta time.
   * @type {number}
   * @readonly
   * @example <caption>Creating an FPS counter</caption>
   * // Create the game object
   * class FPSCounter extends GameObject {
   *   create() {
   *     // Add a 'text' graphics
   *     this.set('text', new Text(`FPS: ${Time.FPS}`));
   *   }
   * 
   *   update() {
   *     // Update the displayed FPS value every frame
   *     this.get('text').text = `FPS: ${Time.FPS}`;
   *   }
   * }
   * 
   * // Add it to a set
   * myScene.set('fps counter', new FPSCounter(myScene));
   */
  static get FPS() {
    return 1000 / Time._deltaTime;
  }

  /**
   * The deltaTime in second, easy to use for animations (e.g myRect.x += 10 * Time.second).
   * @type {number}
   * @readonly
   * @example <caption>Making animations using the delta second</caption>
   * // In some game object's update method, with some graphics in it
   * this.graphics.rect.x += 10 * Time.second; // Move the rect 10px to the right, per second
   * this.graphics.circle.radius += 2 * Time.second; // Make the circle 4px larger every second. 
   */
  static get second() {
    return Time._deltaTime / 1000;
  }
}

/**
 * The delta time.
 * @type {number}
 * @private
 */
Time._deltaTime = 0;

/**
 * The time.
 * @type {number}
 * @private
 */
Time._time = 0;

module.exports = Time;
